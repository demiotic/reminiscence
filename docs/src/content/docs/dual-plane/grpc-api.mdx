This guide covers using Reminiscence's gRPC API for remote access, enabling distributed caching across multiple services, languages, and infrastructure.

## Overview

The gRPC API provides:

- **Remote Access**: Cache accessible over network from any service
- **Multi-Language Support**: Python client included, protobuf enables any language
- **High Performance**: Binary protocol with efficient serialization
- **Production-Ready**: Health checks, metrics, and server reflection
- **Distributed Architecture**: Single cache instance serves multiple clients

## Starting a gRPC Server

### Basic Server

Create a gRPC server with default settings:

```python
from reminiscence import Reminiscence, ReminiscenceConfig
from reminiscence.api.server import create_server

# Create cache instance
config = ReminiscenceConfig(
    db_uri="./cache.db",  # Persistent storage
    max_entries=10000,
    enable_metrics=True,
)
cache = Reminiscence(config)

# Create and start gRPC server
server = create_server(
    cache=cache,
    port=50051,  # Default gRPC port
    max_workers=10,
    enable_reflection=True,  # Enable grpcurl/grpcui
)

server.start()

print("✓ gRPC server running on port 50051")

# Keep server running
try:
    server.wait_for_termination()
except KeyboardInterrupt:
    server.stop(grace=5.0)
    print("Server stopped")
```

### Batch Operations

Process multiple queries efficiently:

```python
from reminiscence.types import (
    LookupRequest,
    StoreRequest,
    MultiModalInput,
    QueryMode,
)

# Batch lookup (3-5x faster than loop)
lookup_requests = [
    LookupRequest(
        query=MultiModalInput(text="What is AI?"),
        context={"model": "gpt-4"},
        mode=QueryMode.AUTO,
    ),
    LookupRequest(
        query=MultiModalInput(text="What is ML?"),
        context={"model": "gpt-4"},
        mode=QueryMode.AUTO,
    ),
    LookupRequest(
        query=MultiModalInput(text="What is DL?"),
        context={"model": "gpt-4"},
        mode=QueryMode.AUTO,
    ),
]

results = client.lookup_batch(lookup_requests)

for i, result in enumerate(results):
    if result.is_hit:
        print(f"Query {i}: ✓ Hit (similarity: {result.similarity:.3f})")
    else:
        print(f"Query {i}: ✗ Miss")

# Batch store
store_requests = [
    StoreRequest(
        query=MultiModalInput(text=f"Query {i}"),
        context={"model": "gpt-4"},
        result=f"Result {i}",
        ttl_seconds=3600,
    )
    for i in range(10)
]

success_flags = client.store_batch(store_requests)
print(f"Stored {sum(success_flags)}/{len(store_requests)} entries")
```

## Error Handling

### Client Errors

Handle connection and timeout errors:

```python
from grpc import RpcError
from reminiscence.api.client import ReminiscenceClient

try:
    with ReminiscenceClient("localhost:50051", timeout=5.0) as client:
        result = client.lookup(query, context)
except RpcError as e:
    if e.code() == grpc.StatusCode.UNAVAILABLE:
        print("Server unavailable - check if server is running")
    elif e.code() == grpc.StatusCode.DEADLINE_EXCEEDED:
        print("Request timeout - server may be overloaded")
    else:
        print(f"gRPC error: {e.details()}")
except Exception as e:
    print(f"Unexpected error: {e}")
```

### Server Errors

Proper error handling in server:

```python
from reminiscence import Reminiscence
from reminiscence.api.server import create_server

try:
    config = ReminiscenceConfig(db_uri="./cache.db")
    cache = Reminiscence(config)
    server = create_server(cache=cache, port=50051)
    server.start()

    print("✓ Server started successfully")
    server.wait_for_termination()

except Exception as e:
    print(f"✗ Server error: {e}")
    # Cleanup

        server.stop(grace=2.0)
    if 'cache' in locals():

```

### Error Result Caching

Control whether to cache error responses:

```python
# By default, errors are NOT cached
try:
    result = expensive_api_call()
    client.store(query, context, result)  # Cached
except APIError as e:
    error_result = {"error": str(e)}
    # This won't be cached (silently skipped)
    client.store(query, context, error_result)

# Explicitly cache errors (e.g., for rate limiting)
client.store(
    query=query,
    context=context,
    result={"error": "Rate limit exceeded"},
    allow_errors=True  # Cache this error
)
```

## Production Patterns

### Pattern 1: Microservices Cache

### Pattern 2: Load-Balanced Cache

Multiple cache servers with client-side load balancing:

```python
import random
from reminiscence.api.client import ReminiscenceClient

# Pool of cache servers
CACHE_SERVERS = [
    "cache-1.example.com:50051",
    "cache-2.example.com:50051",
    "cache-3.example.com:50051",
]

def get_cache_client() -> ReminiscenceClient:
    """Get client connected to random cache server."""
    server = random.choice(CACHE_SERVERS)
    return ReminiscenceClient(server)

# Use consistent hashing for session affinity
import hashlib

def get_cache_client_consistent(key: str) -> ReminiscenceClient:
    """Get client using consistent hashing."""
    hash_value = int(hashlib.md5(key.encode()).hexdigest(), 16)
    server_idx = hash_value % len(CACHE_SERVERS)
    return ReminiscenceClient(CACHE_SERVERS[server_idx])
```

### Pattern 3: Circuit Breaker

Graceful degradation when cache is unavailable:

```python
from reminiscence.api.client import ReminiscenceClient
import grpc

class CacheCircuitBreaker:
    def __init__(self, server: str, failure_threshold: int = 5):
        self.server = server
        self.failure_count = 0
        self.failure_threshold = failure_threshold
        self.is_open = False

    def lookup(self, query, context):
        if self.is_open:
            print("Circuit breaker OPEN - bypassing cache")
            return None  # Cache miss

        try:
            with ReminiscenceClient(self.server, timeout=2.0) as client:
                result = client.lookup(query, context)
                self.failure_count = 0  # Reset on success
                return result if result.is_hit else None

        except grpc.RpcError:
            self.failure_count += 1
            if self.failure_count >= self.failure_threshold:
                self.is_open = True
                print(f"Circuit breaker OPENED after {self.failure_count} failures")
            return None

    def store(self, query, context, result):
        if self.is_open:
            return  # Skip store when circuit is open

        try:
            with ReminiscenceClient(self.server, timeout=2.0) as client:
                client.store(query, context, result)
        except grpc.RpcError:
            pass  # Fail silently for stores

# Usage
cache_breaker = CacheCircuitBreaker("cache-server:50051")

def query_handler(query: str):
    # Try cache
    cached = cache_breaker.lookup(query, {"service": "api"})
    if cached:
        return cached

    # Compute
    result = expensive_computation(query)

    # Store
    cache_breaker.store(query, {"service": "api"}, result)

    return result
```

```python
            print(f"Total Misses: {stats['misses']:,}")

            if stats.get('evictions', 0) > 0:
                print(f"⚠ Evictions: {stats['evictions']:,}")

            if stats.get('lookup_errors', 0) > 0:
                print(f"⚠ Lookup Errors: {stats['lookup_errors']}")

        except Exception as e:
            print(f"✗ Monitoring error: {e}")

        time.sleep(interval)

# Run monitor
monitor_cache("cache-server:50051", interval=30)
```

## Performance Tuning

### Server Tuning

Optimize server for your workload:

```python
server = create_server(
    cache=cache,
    port=50051,
    max_workers=100,  # More workers for high concurrency
    max_message_length=50 * 1024 * 1024,  # 50MB messages
    options=[
        ('grpc.keepalive_time_ms', 10000),
        ('grpc.keepalive_timeout_ms', 5000),
        ('grpc.http2.max_pings_without_data', 0),
        ('grpc.http2.min_time_between_pings_ms', 10000),
        ('grpc.http2.min_ping_interval_without_data_ms', 5000),
    ]
)
```

### Client Tuning

Optimize client connections:

```python
client = ReminiscenceClient(
    address="cache-server:50051",
    timeout=5.0,  # Adjust based on latency
    compression=True,  # Enable for large messages
    options=[
        ('grpc.keepalive_time_ms', 10000),
        ('grpc.max_receive_message_length', 50 * 1024 * 1024),
    ]
)
```

### Batch Operations

Use batch operations for better performance:

```python
# Instead of loop (slow)
for query in queries:
    result = client.lookup(query, context)

# Use batch (3-5x faster)
requests = [LookupRequest(query=q, context=context) for q in queries]
results = client.lookup_batch(requests)
```
